\begin{frame}{乘方的概念}
    \myblock{}{
        一般地，n个相同的因数$a$相乘，记作$a^n$，读作“$a$的n次幂（或$a$的n次方）”，即\\
        
        \bmc{\underbrace{
                \bm{a} \cdot \bm{a} \cdot \bm{a} \cdot \ldots \cdot \bm{a}}_{n\ times} = a^n
            }
    }

\end{frame}
